📖 Como o código funciona

// ARRAY — Lista de valores guardados em uma variável
// Usa colchetes [] com itens separados por vírgula
const cores = [
  "#e63946", "#457b9d", "#2a9d8f",
  "#e9c46a", "#f4a261", "#264653"
];
// cores[0] → "#e63946" (índices começam em 0!)
// getElementById → seleciona pelo id="" exato do HTML
// querySelector  → seleciona como no CSS (mais flexível)
const btn   = document.getElementById('btnTrocar');
const texto = document.getElementById('corAtual');
// Com querySelector seria: document.querySelector('#btnTrocar')
// Math.random() → decimal entre 0.000 e 0.999
// Math.floor()  → arredonda para BAIXO (ex: 4.9 → 4)
// cores.length  → quantidade de itens no array
const i = Math.floor(Math.random() * cores.length);
// Exemplo: 0.73 × 6 = 4.38 → Math.floor = 4 → cores[4]
// addEventListener escuta o clique e dispara a função
// document.body.style → acessa os estilos CSS do body
btn.addEventListener('click', function() {
  const cor = cores[Math.floor(Math.random() * cores.length)];
  document.body.style.backgroundColor = cor;
  texto.textContent = cor;
});
